44  /**
45   * the space on which to drop things
46   * 
47   * @author wf
48   *
49   */
50  public class DropTarget extends Pane {
51    // prepare a LOGGER
52    protected static Logger LOGGER = Logger.getLogger("com.bitplan.dragtop");
53    public static boolean debug = false;
54  
55    public static Color DRAG_ENTERED_COLOR = Color.LIGHTGREEN;
56    public static Color DRAG_DONE_COLOR = Color.LIGHTGRAY;
57    public static Color DRAG_DROPPED_COLOR = Color.LIGHTGRAY;
58    private DropTarget target;
59    List<DragItem> dragItems = new ArrayList<DragItem>();
60  
61    /**
62     * create a Drag Space
63     */
64    public DropTarget() {
65      target = this;
66      target.setOnDragDetected(onDragDetected);
67      target.setOnDragOver(onDragOver);
68      target.setOnDragEntered(onDragEntered);
69      target.setOnDragExited(onDragExited);
70      target.setOnDragDropped(onDragDropped);
71      target.setOnDragDone(onDragDone);
72    }
73  
74    EventHandler<MouseEvent> onDragDetected = new EventHandler<MouseEvent>() {
75      public void handle(MouseEvent event) {
76        /* drag was detected, start drag-and-drop gesture */
77        if (debug) {
78          LOGGER.log(Level.INFO, "onDragDetected");
79        }
80  
81        /* allow any transfer mode */
82        // Dragboard db = target.startDragAndDrop(TransferMode.ANY);
83  
84        /* put a string on dragboard */
85        // ClipboardContent content = new ClipboardContent();
86        // content.putString(target.getText());
87        // db.setContent(content);
88  
89        event.consume();
90      }
91    };
92  
93    EventHandler<DragEvent> onDragOver = new EventHandler<DragEvent>() {
94      public void handle(DragEvent event) {
95        Dragboard db = event.getDragboard();
96        /* data is dragged over the target */
97        if (debug) {
98          LOGGER.log(Level.INFO,
99              String.format("onDragOver %s", db.getContentTypes()));
100       }
101       /*
102        * accept it only if it is not dragged from the same node and if it has a
103        * string data
104        */
105       if (event.getGestureSource() != target) {
106         // && db.hasFiles()
107         /* allow for both copying and moving, whatever user chooses */
108         event.acceptTransferModes(TransferMode.COPY);
109       } else {
110         event.consume();
111       }
112     }
113   };
114 
115   EventHandler<DragEvent> onDragEntered = new EventHandler<DragEvent>() {
116     public void handle(DragEvent event) {
117       /* the drag-and-drop gesture entered the target */
118       if (debug) {
119         LOGGER.log(Level.INFO, "onDragEntered");
120       }
121       /* show to the user that it is an actual gesture target */
122       if (event.getGestureSource() != target) {
123         target.setFill(DRAG_ENTERED_COLOR);
124       } else {
125         target.setFill(Color.LIGHTBLUE);
126       }
127 
128       event.consume();
129     }
130   };
131 
132   EventHandler<DragEvent> onDragExited = new EventHandler<DragEvent>() {
133     public void handle(DragEvent event) {
134       /* mouse moved away, remove the graphical cues */
135       target.setFill(DRAG_DONE_COLOR);
136       event.consume();
137     }
138   };
139 
140   EventHandler<DragEvent> onDragDropped = new EventHandler<DragEvent>() {
141     public void handle(DragEvent event) {
142       /* data dropped */
143       if (debug) {
144         LOGGER.log(Level.INFO, "onDragDropped");
145       }
146       /* if there is file data on dragboard, read it and use it */
147       Dragboard db = event.getDragboard();
148       boolean success = false;
149       if (db.hasFiles()) {
150         success = true;
151         for (File file : db.getFiles()) {
152           DragItem dragItem = new DragItem(file);
153           dragItems.add(dragItem);
154           target.getChildren().add(dragItem);
155           dragItem.setLayoutX(event.getX());
156           dragItem.setLayoutY(event.getY());
157           List<DropHandler> dropHandlers = pluginManager
158               .getExtensions(DropHandler.class);
159           if (dropHandlers.size() > 0)
160             for (DropHandler dropHandler : dropHandlers) {
161               dropHandler.getHandler().accept(dragItem);
162             }
163         }
164         setFill(DRAG_DROPPED_COLOR);
165       }
166       /*
167        * let the source know whether the string was successfully transferred and
168        * used
169        */
170       event.setDropCompleted(success);
171 
172       event.consume();
173     }
174   };
175 
176   EventHandler<DragEvent> onDragDone = new EventHandler<DragEvent>() {
177     public void handle(DragEvent event) {
178       /* the drag-and-drop gesture ended */
179       if (debug) {
180         LOGGER.log(Level.INFO, "onDragDone");
181       }
182       /* if the data was successfully moved, clear it */
183       if (event.getTransferMode() == TransferMode.MOVE) {
184         // target.setText("");
185       }
186       target.setFill(DRAG_DONE_COLOR);
187       event.consume();
188     }
189   };

191   DefaultPluginManager pluginManager;
192 
193   /**
194    * helper to get RGBCode from Color
195    * https://stackoverflow.com/questions/17925318/how-to-get-hex-web-string-from-javafx-colorpicker-color
196    * 
197    * @param color
198    * @return - the webhex
199    */
200   public static String toRGBCode(Color color) {
201     return String.format("#%02X%02X%02X", (int) (color.getRed() * 255),
202         (int) (color.getGreen() * 255), (int) (color.getBlue() * 255));
203   }
204 
205   protected void setFill(Color color) {
206     setStyle(String.format("-fx-background-color: %s;", toRGBCode(color)));
207   }

209   /**
210    * activate the given plugins
211    * 
212    * @param plugins
213    */
214   public void activatePlugins(String plugins) {
215     // https://github.com/pf4j/pf4j/issues/209
216     pluginManager = new JarPluginManager(this.getClass().getClassLoader());
217 
218     if (plugins != null) {
219 
220       for (String plugin : plugins.split(",")) {
221         Path pluginPath = Paths.get(plugin).toAbsolutePath().normalize();
222         pluginManager.loadPlugin(pluginPath);
223       }
224     }
225     pluginManager.startPlugins();
226     showStartedPlugins();
227   }
228 
229   /**
230    * show the started plugins
231    */
232   public void showStartedPlugins() {
233 
234     List<PluginWrapper> startedPlugins = pluginManager.getStartedPlugins();
235 
236     for (PluginWrapper plugin : startedPlugins) {
237       PluginDescriptor descriptor = plugin.getDescriptor();
238       String pluginId = plugin.getDescriptor().getPluginId();
239       String msg = String.format(
240           "Extensions added by plugin id:'%s' version:'%s' %s:", pluginId,
241           descriptor.getVersion(), descriptor.getPluginDescription());
242       if (debug)
243         LOGGER.log(Level.INFO, msg);
244       Set<String> extensionClassNames = pluginManager
245           .getExtensionClassNames(pluginId);
246       for (String extension : extensionClassNames) {
247         msg = "   " + extension;
248         if (debug)
249           LOGGER.log(Level.INFO, msg);
250       }
251     }
252   }
